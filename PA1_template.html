data_fill <- data_fill %>% mutate (day_type = "weekday") 
for (i in 1: nrow (data_fill))
        {
        if (as.POSIXlt (as.Date (data_fill$date[i]))$wday %in% c(6, 0))

                data_fill$day_type[i] <- c("weekend")
